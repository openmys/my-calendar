  useReactCalendar,

} from '@openmys/my-calendar';

function MyCalendar() {

    features: [DateSelection, RangeSelection],
    currentDate: new Date(),
  });

  const month = calendar.getMonth();
  const weeks = month.getWeeks();

  return (
    <div>
      {weeks.map(week => (
        <div key={week.id} className='week'>
          {week.getDays().map(day => (
            <div
              key={day.id}
              onClick={() => day.toggleSelected?.()}
              style={{
                backgroundColor: day.getIsSelected?.() ? 'blue' : 'white',
              }}
            >
              {day.getValue().getDate()}
            </div>
          ))}
        </div>
      ))}
    </div>

}
